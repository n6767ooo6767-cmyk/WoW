// Просто добавь сюда 300 слов, и игра готова const wordPool = ["КАТАР", "МОЛОКО", "АРБУЗ", "КОМПЬЮТЕР", "МАШИНА", "СЕРИАЛ", "БАНШИ", "КОД", "ИГРА", "ЛЕВЕЛ", "КРУТО", "ПИСТОЛЕТ", "ПРИКОЛЛЬНО", "ЛИСИЧКА", "ТЕКСТ", "ПРИВЕТ", "СОЗДАНИЕ", "ЛИЧИКО", "НОСОЧКИ", "ДОЛГИ НА 100", "ЛОЖКА"]; let currentLevelIndex = 0; let currentAnswer = []; function loadLevel() { if (currentLevelIndex >= wordPool.length) { alert("Поздравляю! Ты прошел все 300 уровней!"); return; } const word = wordPool[currentLevelIndex]; // Перемешиваем буквы для интереса const letters = word.split('').sort(() => Math.random() - 0.5); const container = document.getElementById('letters'); container.innerHTML = ''; letters.forEach(char => { const btn = document.createElement('button'); btn.innerText = char; btn.onclick = () => { currentAnswer.push(char); document.getElementById('answer').innerText = currentAnswer.join(''); btn.style.visibility = "hidden"; // "Использовано" }; container.appendChild(btn); }); } // Теперь функция проверки просто сравнивает с оригиналом function checkWord() { const word = currentAnswer.join(''); if (word === wordPool[currentLevelIndex]) { alert("Верно! Уровень " + (currentLevelIndex + 1) + " пройден."); currentLevelIndex++; currentAnswer = []; document.getElementById('answer').innerText = ""; loadLevel(); } else { alert("Неверно! Попробуй еще."); // Сброс currentAnswer = []; document.getElementById('answer').innerText = ""; document.querySelectorAll('#letters button').forEach(b => b.style.visibility = "visible"); } } loadLevel();
